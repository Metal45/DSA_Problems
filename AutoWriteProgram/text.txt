#include <stdio.h>
#define STACK_MAX_SIZE 20
char stack[STACK_MAX_SIZE];
int top=-1;
int isEmpty() { 
if(top<0)
return 1;
else
return 0;
}
void push(int x) { 
if(top==STACK_MAX_SIZE-1)
{ 
printf("Stack is overflow.\n");
}else
{ 
stack[++top]=x;
}
}
int pop() { 
if(top<0)
{ 
printf("Stack is empty'\n");
}
else
return stack[top--];
}
